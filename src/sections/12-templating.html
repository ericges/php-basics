PHP Templating Syntax

Alternative Control Syntax

<?php
// In template files, use the colon syntax — easier to read inside HTML

$isLoggedIn = true;
$items = ['Laptop', 'Mouse', 'Keyboard'];
?>

<?php if ($isLoggedIn): ?>
    <p>Welcome back!</p>
<?php else: ?>
    <p>Please log in.</p>
<?php endif; ?>

<ul>
<?php foreach ($items as $item): ?>
    <li><?= htmlspecialchars($item, ENT_QUOTES, 'UTF-8') ?></li>
<?php endforeach; ?>
</ul>

Available for: if/endif, for/endfor, foreach/endforeach, while/endwhile, switch/endswitch

The Short Echo Tag <?= ?>

<?php
$title = 'My Page';
$user  = 'Alice';
?>

<!DOCTYPE html>
<html>
<head>
    <title><?= htmlspecialchars($title, ENT_QUOTES, 'UTF-8') ?></title>
</head>
<body>
    <h1>Hello, <?= htmlspecialchars($user, ENT_QUOTES, 'UTF-8') ?>!</h1>
</body>
</html>
<?= $var ?> is short for <?php echo $var; ?>. In template files (mixing PHP + HTML), always use the closing ?>. Always wrap output in htmlspecialchars().

A Complete Template

<?php
$pageTitle = 'Product List';
$products  = [
    ['name' => 'Laptop',   'price' => 999.00, 'inStock' => true],
    ['name' => 'Mouse',    'price' =>  29.99, 'inStock' => true],
    ['name' => 'Keyboard', 'price' =>  79.50, 'inStock' => false],
];
?>
<!DOCTYPE html>
<html lang="en">
<head><title><?= htmlspecialchars($pageTitle, ENT_QUOTES, 'UTF-8') ?></title></head>
<body>
  <h1><?= htmlspecialchars($pageTitle, ENT_QUOTES, 'UTF-8') ?></h1>

  <?php if (count($products) > 0): ?>
    <table>
      <tr><th>Product</th><th>Price</th><th>Status</th></tr>
      <?php foreach ($products as $product): ?>
        <tr>
          <td><?= htmlspecialchars($product['name'], ENT_QUOTES, 'UTF-8') ?></td>
          <td>€<?= number_format($product['price'], 2) ?></td>
          <td><?= $product['inStock'] ? 'Available' : 'Sold out' ?></td>
        </tr>
      <?php endforeach; ?>
    </table>
  <?php else: ?>
    <p>No products found.</p>
  <?php endif; ?>
</body>
</html>
