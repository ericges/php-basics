Docblocks

Docblocks start with /** and describe code with annotations like @param, @return or @throws.
IDEs, static analysers, and documentation generators rely on them, e.g., to assist with code understanding and documentation.

<?php

/**
 * Calculate the total price including tax.
 *
 * @param  float  $price    Base price before tax
 * @param  float  $taxRate  Tax rate as a decimal (e.g. 0.19)
 * @return float            Final price with tax applied
 */
function totalWithTax(float $price, float $taxRate): float
{
    return $price * (1 + $taxRate);
}

/**
 * Summarize a list of items.
 *
 * @param  string[]  $items          List of item names
 * @param  int       $limit          Max items to display
 * @return string                    Comma-separated summary
 * @throws InvalidArgumentException  If $items is empty
 */
function summarize(array $items, int $limit = 5): string
{
    if (count($items) < 1) {
        throw new InvalidArgumentException('Items array cannot be empty');
    }

    return implode(', ', array_slice($items, 0, $limit));
}
